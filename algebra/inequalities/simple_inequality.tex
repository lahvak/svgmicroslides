\documentclass[dvisvgm]{standalone}
\usepackage{tikz}
\usepackage{simplesvgpres}
\usetikzlibrary{matrix, fit, decorations.pathreplacing, positioning, calc}

% Simple inequality

\begin{document}
\begin{tikzpicture}[]
    \begin{scope}[svgcls=layer1,
        node distance=0pt,
        every node/.style={inner xsep=0pt}]
        \matrix (line-1) [matrix of math nodes] {3x + 5 & {}>{} & -2\\};
    \end{scope}
    \begin{scope}[svgcls=layer3,
        node distance=0pt,
        every node/.style={inner xsep=0pt}]
        \matrix (line-2) [matrix of math nodes, below=1ex of line-1-1-2.south,
        anchor=line-2-1-2.north] {3x & {}>{} & -7\\};
    \end{scope}
    \begin{scope}[svgcls=layer5,
        node distance=0pt,
        every node/.style={inner xsep=0pt}]
        \matrix (line-3) [matrix of math nodes, below=1ex of line-2-1-2.south,
        anchor=line-3-1-2.north] {x & {}>{} & -\frac{7}{3}\\};
    \end{scope}
    \begin{scope}[svgcls=line, shift={($(line-3-1-2.south) + (3,-3ex)$)}, scale=2]
        \draw[thick, <->] (-4.5,0) -- (1.5,0);
        \foreach \x in {-4,...,1}{
            \draw[thin] (\x,.04) -- (\x,-.04) node[below]{$\x$};
        }
    \end{scope}
    \begin{scope}[svgcls=thirds, shift={($(line-3-1-2.south) + (3,-3ex)$)}, scale=2]
        \draw[red] (-2.33,.03) -- (-2.33,-.03)node[below]{\small$-2\frac{1}{3}$};
        \draw[red] (-2.67,.03) -- (-2.67,-.03)node[below]{\small$-2\frac{2}{3}$};
    \end{scope}
    \begin{scope}[svgcls=right, shift={($(line-3-1-2.south) + (3,-3ex)$)}, scale=2]
        \draw[blue, very thick, ->] (-2.33,0) -- (1.5,0);
    \end{scope}
    \begin{scope}[svgcls=circ, shift={($(line-3-1-2.south) + (3,-3ex)$)}, scale=2]
        \draw[blue, thick, fill=white] (-2.33,0) circle[radius=1pt];
    \end{scope}
    \node[fit=(line-1) (line-2) (line-3)] (bbox) {};
    \begin{scope}[svgcls=layer2, decoration=brace]
        \node[below=of bbox, align=left] {Subtract 5 from both sides: $-2 - 5 =
        -7$.};
    \end{scope}
    \begin{scope}[svgcls=layer4]
        \node[below=of bbox, align=left] {Divide both sides by $3$.};
    \end{scope}
    \begin{scope}[svgcls=layer6]
        \node[below=of bbox, align=left] {Solutions are all numbers strictly
            greater than $-\frac{7}{3}$.};
    \end{scope}
    \begin{scope}[svgcls=layer8]
        \node[below=of bbox, align=left] {$-\frac{7}{3} = -2\frac{1}{3}$};
    \end{scope}
    \begin{scope}[svgcls=layer10]
        \node[below=of bbox, align=left] {Draw a part of the number line with
            $-2\frac{1}{3}$ roughly in the middle.};
    \end{scope}
    \begin{scope}[svgcls=layer12]
        \node[below=of bbox, align=left] {$-2\frac{1}{3}$ is third of the way
            from $-2$ to $-3$.};
    \end{scope}
    \begin{scope}[svgcls=layer14]
        \node[below=of bbox, align=left] {Larger numbers are to the
            \textbf{right} of $-2\frac{1}{3}$.};
    \end{scope}
    \begin{scope}[svgcls=layer16]
        \node[below=of bbox, align=left] {The inequality is sharp, so
            $-2\frac{1}{3}$ is not included.\\We indicate this by an empty
            circle.};
    \end{scope}
    \definePages{
        {layer1},

        {layer1, layer3, layer4},
        {layer1, layer3, layer4, layer5},

        {layer1, layer3, layer5, layer8},
        {layer1, layer3, layer5, line, layer10},
        {layer1, layer3, layer5, line, thirds, layer12},
        {layer1, layer3, layer5, line, thirds, right, layer14},
        {layer1, layer3, layer5, line, thirds, right, circ, layer16}}
\end{tikzpicture}
\end{document}
